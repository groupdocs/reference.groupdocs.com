PresentationLoadOptions loadOptions = new PresentationLoadOptions();
using (Watermarker watermarker = new Watermarker(@"D:\input.ppt", loadOptions))

    // Use el método Agregar para agregar una marca de agua a una diapositiva en particular o a todas las diapositivas.

    // Guardar cambios.
    watermarker.Save(@"D:\output.ppt");
}
