## GetImageStream() {#getimagestream}

이미지 스트림을 반환합니다.

```csharp
public Stream GetImageStream()
```

### 반환 값

이미지가 있는 스트림.

### 예

다음 예에서는 이미지를 파일에 저장하는 방법을 보여줍니다.

```csharp
// Parser 클래스의 인스턴스 생성
using (Parser parser = new Parser(filePath))
{
    // 문서에서 이미지 추출
    IEnumerable<PageImageArea> images = parser.GetImages();
    
    // 이미지 추출이 지원되는지 확인
    if (images == null)
    {
        Console.WriteLine("Page images extraction isn't supported");
        return;
    }

    // 이미지 반복
    foreach (PageImageArea image in images)
    {
        // 이미지 스트림 열기
        using (Stream imageStream = image.GetImageStream())
        {
            // 이미지를 저장할 파일 생성
            using (Stream destStream = File.Create(Guid.NewGuid().ToString() + image.FileType.Extension))
            {
                byte[] buffer = new byte[4096];
                int readed = 0;

                do
                {
                    // 이미지 스트림에서 데이터 읽기
                    readed = imageStream.Read(buffer, 0, buffer.Length);

                    if (readed > 0)
                    {
                        // 파일 스트림에 데이터 쓰기
                        destStream.Write(buffer, 0, readed);
                    }
                }
                while (readed > 0);
            }
        }
    }
}
```

### 반환 값

### 예

다음 예에서는 이미지를 다른 형식의 파일에 저장하는 방법을 보여줍니다.

```csharp
// Parser 클래스의 인스턴스 생성
using (Parser parser = new Parser(filePath))
{
    // 문서에서 이미지 추출
    IEnumerable<PageImageArea> images = parser.GetImages();
    
    // 이미지 추출이 지원되는지 확인
    if (images == null)
    {
        Console.WriteLine("Page images extraction isn't supported");
        return;
    }

    // 이미지를 PNG 형식으로 저장하는 옵션 생성
    ImageOptions options = new ImageOptions(ImageFormat.Png);
    
    // 이미지 반복
    foreach (PageImageArea image in images)
    {
        // 이미지 스트림 열기
        using (Stream imageStream = image.GetImageStream(options))
        {
            // 이미지를 저장할 파일 생성
            using (Stream destStream = File.Create(Guid.NewGuid().ToString() + ".png"))
            {
                byte[] buffer = new byte[4096];
                int readed = 0;

                do
                {
                    // 이미지 스트림에서 데이터 읽기
                    readed = imageStream.Read(buffer, 0, buffer.Length);

                    if (readed > 0)
                    {
                        // 파일 스트림에 데이터 쓰기
                        destStream.Write(buffer, 0, readed);
                    }
                }
                while (readed > 0);
            }
        }
    }
}
```
